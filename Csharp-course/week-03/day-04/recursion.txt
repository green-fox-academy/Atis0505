Factorial with loop

public static int FactorialWithLoop(int n) 
{
    int factorial = 1;
    for (int i = 1; i  < n + 1; i++) 
    {
      factorial *= i;
    }
    return factorial;
}


Factorial with recursion

public static int FactorialWithRecursion(int n) 
{
    if (n == 1) 
    {
        return 1;
    } 
    else 
    {
        return n * FactorialWithRecursion(n - 1);
    }
}
